DECLARE 
    numer_max INT;
    new_department_name VARCHAR2(30) := 'EDUCATION';
BEGIN
    SELECT MAX(department_id) +10
    INTO numer_max
    FROM departments;
    INSERT INTO departments(department_id, department_name)
    VALUES(numer_max, new_department_name);
  END;
